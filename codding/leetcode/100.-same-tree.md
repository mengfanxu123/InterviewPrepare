# 100. Same Tree

```javascript
 const treeNode = (val) = {
 this.val = val;
 this.left = this.right = null;
 }
 
 const SameTree = (p, q) => {
     if (!p && !q) return true;
     if (!p || !q) return false;
     if (p.val === q.val){
     return (SameTree(p.left, q.left) && SameTree(p.right, q.right))
     }
     return false;
 }
```
